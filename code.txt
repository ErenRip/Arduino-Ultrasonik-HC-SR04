#include <LiquidCrystal_I2C.h>

#include <Wire.h>

LiquidCrystal_I2C lcdekranim(0x27,16,2);

void setup() {

pinMode(4,OUTPUT);  
Serial.begin(9600);

lcdekranim.init();
lcdekranim.backlight();
lcdekranim.setCursor(0,0);
lcdekranim.print("Giris Ekrani");
delay(1000);

pinMode(6,OUTPUT);//Trig
pinMode(7,INPUT); //Echo


}

void loop() {

  
digitalWrite(6,1); delay(1);
digitalWrite(6,0);
int sure=pulseIn(7,1);
int mesafe=(sure/2)/28.97;
Serial.print("Olculen Mesafe=");
Serial.println(mesafe);
delay(1);


  lcdekranim.clear();
if(mesafe<25)
{
  digitalWrite(4,1);
  }
if(mesafe<=50)
{
  digitalWrite(4,1);
  delay(0020);
  digitalWrite(4,0);
  delay(0020);
  digitalWrite(4,1);
  delay(0020);
  digitalWrite(4,0);
  
 }

else
{
  digitalWrite(4,0);
  } 


}
